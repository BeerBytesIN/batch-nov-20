class GFG {
	public static void main (String[] args) {
		int j=0,temp=0;
		int[] arr = {-12, 11, -13, -5, 6, -7, 5, -3, -6};
	    for(int i=0;i<arr.length;i++){
	        if(arr[i]<0){
	            if(i!=j){
    	            temp = arr[i];
    	            arr[i] = arr[j];
    	            arr[j] = temp;
	            }
	            j++;
	        }
	    }	
	    System.out.println(arr);
	}
}
